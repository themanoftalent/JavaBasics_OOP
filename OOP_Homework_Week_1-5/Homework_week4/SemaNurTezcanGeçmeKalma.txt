public class Not{

    int toplam= 0;
    	
    System.out.println("5 adet sınav notunuzu giriniz: ");
    	
    for(int i=0; i<5; i++) {
    	int not= sc.nextInt();
    	toplam+= not;    		
    }

    }
    
}
